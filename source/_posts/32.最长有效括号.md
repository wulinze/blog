### 最长有效括号

主要目的是选择子串匹配最长的有效括号。

* 方法一：DP

~~~c++
class Solution {
public:
    int longestValidParentheses(string s) {
        if(s.length() == 0)return 0;
        int res = 0;
        vector<int> dp(s.length()+1, 0);

        for(int i=1; i<s.length(); i++){
            if(s[i] == ')'){
                if(s[i-1] == '('){
                    dp[i+1] = dp[i-1]+2;
                } else if(i-dp[i]-1 >=0 && s[i-dp[i]-1]=='('){
                    dp[i+1] = dp[i]+dp[i-dp[i]-1]+2;
                }
                
                res = max(res, dp[i+1]);
            }
            
        }

        return res;
    }
};
~~~

* 方法二：栈

~~~c++
class Solution {
public:
    int longestValidParentheses(string s) {
        if(s.length() == 0)return 0;
        int res = 0;
        stack<int> pos;
        pos.push(-1);

        for(int i=0; i<s.length(); i++){
            if(s[i] == '(')pos.push(i);
            else if(s[i] == ')'){
                if(pos.top() == -1 || s[pos.top()] == ')')pos.push(i);
                else {pos.pop();res = max(res, i-pos.top());}
            }
        }

        return res;
    }
};
~~~

* 方法三：巧妙对接

~~~c++
class Solution {
public:
    int longestValidParentheses(string s) {
        if(s.length() == 0)return 0;
        int res = 0;
        int  left=0, right=0;
        
        for(auto& c : s){
            if(c == '(')left++;
            if(c == ')')right++;
            if(right > left){
                left = right = 0;
            } else if(left == right){
                res = max(res, 2*right);
            }
        }

        right = left = 0;
        for(int i=s.length()-1; i>=0; i--){
            if(s[i] == ')')left++;
            if(s[i] == '(')right++;
            if(right > left){
                left = right = 0;
            } else if(left == right){
                res = max(res, 2*right);
            }
        }

        return res;
    }
};
~~~
