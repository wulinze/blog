### 617 .合并二叉树(简单)

* 题目描述:给定两个二叉树，想象当你将它们中的一个覆盖到另一个上时，两个二叉树的一些节点便会重叠。

  你需要将他们合并为一个新的二叉树。合并的规则是如果两个节点重叠，那么将他们的值相加作为节点合并后的新值，否则不为 NULL 的节点将直接作为新二叉树的节点。

  ![1600830004885](.\images\1600830004885.png)

* 前序遍历+递归

~~~c++
TreeNode* mergeTrees(TreeNode* t1, TreeNode* t2) {
        if(!t1) {
            return t2;
        }

        if(!t2) {
            return t1;
        }
        t1->val = t1->val + t2->val;
        t1->left = mergeTrees(t1->left, t2->left);
        t1->right = mergeTrees(t1->right, t2->right);

        return t1;
    }
~~~

* 前序遍历+迭代

~~~c++
TreeNode* mergeTrees(TreeNode* t1, TreeNode* t2) {
        if(!t1)return t2;
        stack<TreeNode*> s1, s2;
        TreeNode *p1(t1), *p2(t2);
        pair<TreeNode*, bool> tmp(p1 , false);

        while(!s1.empty() || p1 || !s2.empty() || p2){
            if(!p2){
                if(!s1.empty()){
                    p1 = s1.top(); s1.pop();
                    tmp = {p1, false};
                    p1 = p1->right;
                }
                if(!s2.empty()){
                    p2 = s2.top(); s2.pop();
                    p2 = p2->right;
                }else break;
            }else if(!p1){
                if(tmp.second){
                    tmp.first->left = new TreeNode(p2->val);
                    p1 = tmp.first->left;
                }else {
                    tmp.first->right = new TreeNode(p2->val);
                    p1 = tmp.first->right;
                }
                s1.push(p1);s2.push(p2);
                tmp = {p1, true};
                p2 = p2->left;
                p1 = p1->left;
                continue;
            }else {
                s1.push(p1);s2.push(p2);
                p1->val += p2->val;
                tmp = {p1, true};
                p1 = p1->left; p2 = p2->left;
            }
        }

        return t1;
    }
~~~
